Django
1. python -m venv myenv (close and reopen vs code)
2. Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser
3. myenv\Scripts\activate  
4. cd .\backend\
5. pip install django (close and reopen vs code)
6. pip install virtualenv 
7. pip install djangorestframework
8. python -m pip install django  
9. py -m pip install --upgrade pip      
10. python -m pip install django-cors-headers
11. pip install pillow
12. python manage.py makemigrations
13. python manage.py migrate  
14. python manage.py runserver
15. google link: http://127.0.0.1:8000/api/
